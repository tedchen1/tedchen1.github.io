sql学习

/INSERTINTO语句/
create table Test(id int,name char(20),sex char(10)) /创建表/
insert into Test values(01,’ted’,’男’) /插入数据/
insert into Test values(02,’make’,’男’)
insert into Test values(03,’lina’,’女’)
insert into Test values(05,’lina’,’女’)
insert into Test (id) values(04) /对应插入数据/

/UPDATE表名称SET列名称=新值WHERE列名称=某值/
update Test set id=1where name=’ted’
update Test set name=’kimi’,sex=’女’where id=4 /更新某一行中的若干列/

/DELETE语句 DELETE语句用于删除表中的行。/
delete from Test where id=5

select top 2 from Test
select top 50 percent from Test/访问前50%/

/Like 操作/
select from Test where name like ‘%t%’
select from Test where name not like ‘%t%’/not like/

/IN 操作/
select from Test where name IN(‘make’,’lina’)
select from Test where name =’make’ or name=’lina’

/between…and操作/
select * from Test where name between ‘make’ and ‘ted’

/使用表别名/
select a.name,a.sex,a.id from Test as a where a.name=’ted’

create table Test1(sid int,score int)
insert into Test1 values(4,100)

/引用两个表/
select a.id,a.name,a1.score from Test as a, Test1 as a1 where a.id=a1.sid
select a.id,a.name,a1.score from Test as a inner join Test1 as a1 on a.id=a1.sid /inner join …on..方法/

select name from Test inner join Test1 on Test.id=Test1.sid/两张表有一个匹配就可以用返回数值/
select name from Test left join Test1 on Test.id=Test1.sid
select sid from Test right join Test1 on Test.id=Test1.sid
select name from Test full join Test1 on Test.id=Test1.sid
select from Test1
select from Test
